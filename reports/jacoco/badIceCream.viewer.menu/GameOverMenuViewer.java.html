    @Override
    public void drawElements(Graphics gui) {
        drawTitle(gui);
        drawSnowflake(gui);

        int height = 17;

        for(int i = 0; i < getModel().getNumberEntries(); i++) {
            gui.drawText(new Position(65, height + i), getModel().getEntry(i), getModel().isSelected(i) ? "#D1D100" : "#FFFFFF" );

        }
