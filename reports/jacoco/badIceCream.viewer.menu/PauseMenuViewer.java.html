    @Override
    public void drawElements(Graphics gui) {
        drawTitle(gui);
        drawSnowflake(gui);
        drawPauseSymbol(gui);

        int height = 21;

        for(int i = 0; i < getModel().getNumberEntries(); i++) {
            gui.drawText(new Position(68, height + i), getModel().getEntry(i), getModel().isSelected(i) ? "#D1D100" : "#FFFFFF" );
            height += 2;
        }
    }
